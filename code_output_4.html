# Determine the optimal number of clusters using the Elbow Method and Silhouette Analysis
inertia = []
silhouette_scores = []
K = range(2, 11)

for k in K:
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
    kmeans.fit(mas_data_cleaned[numerical_features])
    inertia.append(kmeans.inertia_)
    silhouette_avg = silhouette_score(mas_data_cleaned[numerical_features], kmeans.labels_)
    silhouette_scores.append(silhouette_avg)

# Plot the Elbow Method results
plt.figure(figsize=(10, 6))
plt.plot(K, inertia, 'bo-')
plt.xlabel('Number of Clusters')
plt.ylabel('Inertia')
plt.title('Elbow Method for Optimal Clusters')
plt.show()

# Plot the Silhouette Scores
plt.figure(figsize=(10, 6))
plt.plot(K, silhouette_scores, 'bo-')
plt.xlabel('Number of Clusters')
plt.ylabel('Silhouette Score')
plt.title('Silhouette Scores for Different Cluster Sizes')
plt.show()

# Assume the optimal number of clusters is chosen based on the above analysis
optimal_clusters = 3
kmeans = KMeans(n_clusters=optimal_clusters, random_state=42, n_init=10)
clusters = kmeans.fit_predict(mas_data_cleaned[numerical_features])

# Add cluster labels to the data
mas_data_cleaned['Cluster'] = clusters

# Calculate the Silhouette Score for the chosen number of clusters
silhouette_avg = silhouette_score(mas_data_cleaned[numerical_features], clusters)
print(f'Silhouette Score for {optimal_clusters} clusters: {silhouette_avg}')
